    11→/**
    12→ * Extract the topic from a user question like "what is a metaphor" or "explain photosynthesis".
    13→ * Strips common question prefixes and returns the core topic.
    14→ */
    15→function extractTopic(question: string): string {
    16→  const lower = question.toLowerCase().trim();
    17→  // Remove common prefixes
    18→  const prefixes = [
    19→    "what is the meaning of ",
    20→    "what is the definition of ",
    21→    "what is the definition ",
    22→    "what is meant by ",
    23→    "what is ",
    24→    "what are ",
    25→    "what's ",
    26→    "define ",
    27→    "explain ",
    28→    "meaning of ",
    29→    "meaning ",
    30→    "can you explain ",
    31→    "please explain ",
    32→    "could you explain ",
    33→    "help me understand ",
    34→    "tell me about ",
    35→    "describe ",
    36→  ];
    37→  let topic = lower;
    38→  for (const prefix of prefixes) {
    39→    if (topic.startsWith(prefix)) {
    40→      topic = topic.slice(prefix.length);
    41→      break;
    42→    }
    43→  }
    44→  // Remove trailing question marks and "in simple terms" etc.
    45→  topic = topic.replace(/\?+.*$/, "").replace(/ in .+terms.*$/, "").replace(/ please.*$/, "").trim();
    46→  // Capitalize first letter for display
    47→  return topic.charAt(0).toUpperCase() + topic.slice(1);
    48→}

    85→  // Normalize the topic for lookup
    86→  const normalizedTopic = topic.toLowerCase().replace(/\s+/g, " ").trim();
    87→
    88→  // Check our known concepts dictionary first
    89→  for (const [key, explanation] of Object.entries(KNOWN_CONCEPTS)) {
    90→    if (normalizedTopic === key || normalizedTopic.includes(key) || key.includes(normalizedTopic)) {
    91→      return explanation;
    92→    }
    93→  }
    94→
    95→  // For unknown topics, generate a helpful response that still addresses the question
    96→  // rather than asking for more clarification
    97→  return `Great question! Let me explain **${topic}** for you.\n\n${topic} is an important concept to understand.    182→/**
   183→ * Simulate streaming by splitting a response into chunks and sending them
   184→ * with small delays - gives the feel of a real AI streaming response.
   185→ */
   186→function createMockStream(responseText: string): ReadableStream {
   187→  const encoder = new TextEncoder();
   188→  const words = responseText.split(/(\s+)/);
   189→  let index = 0;
   190→
   191→  return new ReadableStream({
   192→    async start(controller) {
   193→      const chunkSize = 3;
   194→      const delay = 30;
   195→
   196→      async function sendChunk() {
   197→        if (index >= words.length) {
   198→          controller.enqueue(encoder.encode(sseEvent("[DONE]")));
   199→          controller.close();
   200→          return;
   201→        }
   202→
   203→        const chunk = words.slice(index, index + chunkSize).join("");
   204→        index += chunkSize;
   205→
   206→        if (chunk) {
   207→          controller.enqueue(
   208→            encoder.encode(sseEvent(JSON.stringify({ content: chunk })))
   209→          );
   210→        }
   211→
   212→        await new Promise((resolve) => setTimeout(resolve, delay));
   213→        await sendChunk();
   214→      }
   215→
   216→      await sendChunk();
   217→    },
   218→  });
   219→}

   222→  try {

   224→

   226→      return new Response(
   227→        sseEvent(JSON.stringify({ error: "Messages array is required" })),
   228→        { status: 400, headers: { "Content-Type": "text/event-stream" } }
   229→      );
   230→    }
   231→
   232→    for (const msg of messages) {
   233→      if (!msg.role || typeof msg.content !== "string") {
   234→        return new Response(
   235→          sseEvent(
   236→            JSON.stringify({
   237→              error: "Each message must have a 'role' and 'content' string",
   238→            })
