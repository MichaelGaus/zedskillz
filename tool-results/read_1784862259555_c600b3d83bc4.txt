    15→    15→function extractTopic(question: string): string {
    16→    16→  const lower = question.toLowerCase().trim();
    17→    17→  // Remove common prefixes
    18→    18→  const prefixes = [
    19→    19→    "what is the meaning of ",
    20→    20→    "what is the definition of ",
    21→    21→    "what is the definition ",
    22→    22→    "what is meant by ",
    23→    23→    "what is ",
    24→    24→    "what are ",
    25→    25→    "what's ",
    26→    26→    "define ",
    27→    27→    "explain ",
    28→    28→    "meaning of ",
    29→    29→    "meaning ",
    30→    30→    "can you explain ",
    31→    31→    "please explain ",
    32→    32→    "could you explain ",
    33→    33→    "help me understand ",
    34→    34→    "tell me about ",
    35→    35→    "describe ",
    36→    36→  ];
    37→    37→  let topic = lower;
    38→    38→  for (const prefix of prefixes) {
    39→    39→    if (topic.startsWith(prefix)) {
    40→    40→      topic = topic.slice(prefix.length);
    41→    41→      break;
    42→    42→    }
    43→    43→  }
    44→    44→  // Remove trailing question marks and "in simple terms" etc.
    45→    45→  topic = topic.replace(/\?+.*$/, "").replace(/ in .+terms.*$/, "").replace(/ please.*$/, "").trim();
    46→    46→  // Capitalize first letter for display
    47→    47→  return topic.charAt(0).toUpperCase() + topic.slice(1);
    48→    48→}

   299→   299→function createMockStream(responseText: string): ReadableStream {
   300→   300→  const encoder = new TextEncoder();
   301→   301→  const words = responseText.split(/(\s+)/);
   302→   302→  let index = 0;

   312→   312→          controller.close();
   313→   313→          return;
   314→   314→        }
   315→   315→
   316→   316→        const chunk = words.slice(index, index + chunkSize).join("");
   317→   317→        index += chunkSize;
   318→   318→
   319→   319→        if (chunk) {
   320→   320→          controller.enqueue(

   326→   326→        await sendChunk();

   330→   330→    },

   334→   334→/**
   335→   335→ * Ensure Gemini chat history has proper alternating roles.
   336→   336→ * Gemini requires: first message must be "user", then alternating "user"/"model".
   337→   337→ * This function:
   338→   338→ * 1. Filters out greeting messages that start with "model" role
   339→   339→ * 2. Merges consecutive same-role messages
   340→   340→ * 3. Ensures history starts with "user"
   341→   341→ * 4. Returns the cleaned history + the last user message separately
   342→   342→ */
   343→   343→function sanitizeGeminiHistory(
   344→   344→  messages: { role: string; content: string }[]
   345→   345→): { history: { role: "user" | "model"; parts: { text: string }[] }[]; lastUserMessage: string } {
   346→   346→  // Convert roles
   347→   347→  const converted = messages.map((m) => ({
   348→   348→    role: m.role === "ai" || m.role === "assistant" ? "model" : m.role === "system" ? "system" : "user",
   349→   349→    content: m.content,
   350→   350→  }));
   351→   351→
   352→   352→  // Filter out system messages
   353→   353→  const filtered = converted.filter((m) => m.role !== "system");
   354→   354→
   355→   355→  // Ensure alternation: merge consecutive same-role messages
   356→   356→  const merged: { role: "user" | "model"; content: string }[] = [];
   357→   357→  for (const msg of filtered) {
   358→   358→    if (merged.length > 0 && merged[merged.length - 1].role === msg.role) {
   359→   359→      // Merge with previous message of same role
   360→   360→      merged[merged.length - 1].content += "\n\n" + msg.content;
   361→   361→    } else {
   362→   362→      merged.push({ role: msg.role as "user" | "model", content: msg.content });
   363→   363→    }
   364→   364→  }
   365→   365→
   366→   366→  // Ensure history starts with "user" role — skip initial "model" messages
   367→   367→  // (these are usually the AI greeting that the user didn't explicitly request)
   368→   368→  while (merged.length > 0 && merged[0].role !== "user") {
   369→   369→    merged.shift();
   370→   370→  }
   371→   371→
   372→   372→  // We need at least 2 messages for history (1 user + 1 model)
   373→   373→  // and the last message should be the user's current question
   374→   374→  const lastUserMsg = merged.filter((m) => m.role === "user").pop()?.content || "Hello";
   375→   375→
   376→   376→  // History is everything except the last user message
   377→   377→  const historyMessages = merged.slice(0, -1);
   378→   378→
   379→   379→  const history = historyMessages.map((m) => ({
   380→   380→    role: m.role,
   381→   381→    parts: [{ text: m.content }],
   382→   382→  }));
   383→   383→
   384→   384→  return { history, lastUserMessage: lastUserMsg };
   385→   385→}
